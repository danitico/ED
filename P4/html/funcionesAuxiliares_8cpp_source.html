328  // Vector que tienen el vertice de inicio

335  resultante.addVertex(grafo.currVertex().getData());

340  // Ponemos que su coste es 0
341  coste[grafo.currVertex().getLabel()]=0;
342  // Mientras no tengamos el objetivo, es decir, visitar todos...
343  while(visitados!=objetivo){
344  float minimo=800;
345  int candidato=-1;
346

354  // si vemos que no ha sido visitado
355  if(visitados[a]==0){
356  float coste1=coste[a];

359  // es mejor que su coste anterior (es decir, que sea menor)

397  // Añadimos este vertice al grafo resultante

401 
402

540  }
